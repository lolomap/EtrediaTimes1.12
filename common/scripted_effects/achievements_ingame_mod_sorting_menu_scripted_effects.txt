# usage:
## sort_archievements_hide_unavailable = yes
sort_archievements_hide_unavailable = {
	# Vanilla Achievements:
	clear_array = custom_achievements_array_vanilla_visible
	for_each_loop = {
		array = global.custom_achievements_array_vanilla
		value = v
		set_temp_variable = { achievement_id = v }
		round_temp_variable = achievement_id
		if = {
			limit = {
				meta_trigger = {
					text = {
						available_conditions_met_for_custom_vanilla_achievement_[ENTRY] = yes
					}
					ENTRY = "[?achievement_id]"
				}
			}
			add_to_array = { custom_achievements_array_vanilla_visible = v }
		}
	}
	# Steam Achievements:
	clear_array = custom_achievements_array_steam_visible
	for_each_loop = {
		array = global.custom_achievements_array_steam
		value = v
		set_temp_variable = { achievement_id = v }
		round_temp_variable = achievement_id
		if = {
			limit = {
				meta_trigger = {
					text = {
						available_conditions_met_for_custom_steam_achievement_[ENTRY] = yes
					}
					ENTRY = "[?achievement_id]"
				}
			}
			add_to_array = { custom_achievements_array_steam_visible = v }
		}
	}
}

# usage:
## sort_archievements_hide_completed = yes
sort_archievements_hide_completed = {
	# Vanilla Achievements:
	for_each_loop = {
		array = global.custom_achievements_array_vanilla
		set_temp_variable = { achievement_id = v }
		if = {
			limit = {
				has_unlocked_custom_achievement = yes
				is_in_array = { custom_achievements_array_vanilla_visible = achievement_id }
			}
			remove_from_array = { custom_achievements_array_vanilla_visible = achievement_id }
		}
	}
	# Steam Achievements:
	for_each_loop = {
		array = global.custom_achievements_array_steam
		set_temp_variable = { achievement_id = v }
		if = {
			limit = {
				has_unlocked_custom_achievement = yes
				is_in_array = { custom_achievements_array_steam_visible = achievement_id }
			}
			remove_from_array = { custom_achievements_array_steam_visible = achievement_id }
		}
	}
}

## sort_archievements_show_completed = yes
sort_archievements_show_completed = {
	# Vanilla Achievements:
	for_each_loop = {
		array = global.custom_achievements_array_vanilla
		set_temp_variable = { achievement_id = v }
		if = {
			limit = {
				has_unlocked_custom_achievement = yes
				NOT = { is_in_array = { custom_achievements_array_vanilla_visible = achievement_id } }
			}
			add_to_array = { custom_achievements_array_vanilla_visible = achievement_id }
			if = {
				limit = {
					is_in_array = { custom_achievements_array_vanilla_hidden_manually = achievement_id }
				}
				remove_from_array = { custom_achievements_array_vanilla_hidden_manually = achievement_id }
			}
		}
	}
	# Steam Achievements:
	for_each_loop = {
		array = global.custom_achievements_array_steam
		set_temp_variable = { achievement_id = v }
		if = {
			limit = {
				has_unlocked_custom_achievement = yes
				NOT = { is_in_array = { custom_achievements_array_steam_visible = achievement_id } }
			}
			add_to_array = { custom_achievements_array_steam_visible = achievement_id }
			if = {
				limit = {
					is_in_array = { custom_achievements_array_steam_hidden_manually = achievement_id }
				}
				remove_from_array = { custom_achievements_array_steam_hidden_manually = achievement_id }
			}
		}
	}
}

## sort_archievements_completed_first = yes
sort_archievements_completed_first = {
	# Vanilla Achievements:
	for_each_loop = {
		array = global.custom_achievements_array_vanilla
		set_temp_variable = { achievement_id = v }
		if = {
			limit = {
				has_unlocked_custom_achievement = yes
			}
			if = {
				limit = {
					is_in_array = { custom_achievements_array_vanilla_visible = achievement_id }
				}
				remove_from_array = { custom_achievements_array_vanilla_visible = achievement_id }
			}
			add_to_array = { array = custom_achievements_array_vanilla_visible value = achievement_id index = 0 }
			if = {
				limit = {
					is_in_array = { custom_achievements_array_vanilla_hidden_manually = achievement_id }
				}
				remove_from_array = { custom_achievements_array_vanilla_hidden_manually = achievement_id }
			}
		}
	}
	# Steam Achievements:
	for_each_loop = {
		array = global.custom_achievements_array_steam
		set_temp_variable = { achievement_id = v }
		if = {
			limit = {
				has_unlocked_custom_achievement = yes
			}
			if = {
				limit = {
					is_in_array = { custom_achievements_array_steam_visible = achievement_id }
				}
				remove_from_array = { custom_achievements_array_steam_visible = achievement_id }
			}
			add_to_array = { array = custom_achievements_array_steam_visible value = achievement_id index = 0 }
			if = {
				limit = {
					is_in_array = { custom_achievements_array_steam_hidden_manually = achievement_id }
				}
				remove_from_array = { custom_achievements_array_steam_hidden_manually = achievement_id }
			}
		}
	}
}
