        cost = 75

        complete_effect = {

            }

                for_each_scope_loop = {

                        set_variable_to_random = oppos_@PREV
                        set_variable_to_random = {var = sep_@PREV max = 0.5}

            base = 0
            modifier = {
                check_variable = { ucb_reg_fas > 0.25 }

                add = 500
            }
        }
    }
