defined_text = {
	name = GetCustomAchievementName

	text = {
		trigger = {
			set_temp_variable = { achievement_category = 1 }
			set_temp_variable = { achievement_id = v }
			is_custom_achievement_in_category = yes
		}
		localization_key = "[This.GetCustomAchievementNameVanilla]"
	}
	text = {
		trigger = {
			set_temp_variable = { achievement_category = 2 }
			set_temp_variable = { achievement_id = v }
			is_custom_achievement_in_category = yes
		}
		localization_key = "[This.GetCustomAchievementNameSteam]"
	}
	text = {
		trigger = {
			always = yes
		}
		localization_key = ""
	}
}
